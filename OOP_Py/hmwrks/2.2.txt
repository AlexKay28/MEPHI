Напишите программу на языке Python, читающую со стандартного ввода три целых числа (по одному накаждой строке) и печатающую на стандартный вывод количество вторых по величине.
Комментарий: для чисел 1, 2, 3 ответом будет 1; для чисел 1, 1, 2 — 2; а для чисел 3, 3, 3 — 0.

#Second task
a, b, c = int(input()), int(input()), int(input())
array = [a, b, c]
seq = set(array)
seq.remove(max(set(array)))
if len(seq) == 0:
    print(0)
else:
    Nmax = array.count(max(seq))
    print(Nmax)
